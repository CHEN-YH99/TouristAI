```typescript
// Token刷新
async refreshToken(userId: string): Promise<{ accessToken: string }> {
  const user = await this.userRepository.findOne({ where: { id: userId } });
  if (!user) throw new UnauthorizedException('用户不存在');
  
  const payload = { sub: user.id, email: user.email };
  return {
    accessToken: this.jwtService.sign(payload, { expiresIn: '7d' })
  };
}

// 密码强度验证
private validatePassword(password: string): void {
  if (password.length < 6) {
    throw new BadRequestException('密码至少6位');
  }
  if (!/[a-zA-Z]/.test(password) || !/[0-9]/.test(password)) {
    throw new BadRequestException('密码必须包含字母和数字');
  }
}

// 用户名验证
private validateUsername(username: string): void {
  if (username.length < 2 || username.length > 20) {
    throw new BadRequestException('用户名长度为2-20位');
  }
  if (!/^[a-zA-Z0-9_]+$/.test(username)) {
    throw new BadRequestException('用户名只能包含字母、数字和下划线');
  }
}

// 账号锁定检查
private async checkAccountLock(email: string): Promise<void> {
  const lockKey = `account_lock:${email}`;
  const failCount = await this.redis.get(`login_fail:${email}`);
  
  if (failCount >= 5) {
    const lockTime = await this.redis.ttl(lockKey);
    throw new UnauthorizedException(
      `账号已锁定，请${Math.ceil(lockTime / 60)}分钟后重试`
    );
  }
}
```

```typescript
// 分页查询用户攻略
async findByUserId(
  userId: string, 
  page: number = 1, 
  limit: number = 10
): Promise<{ data: Guide[]; total: number; page: number; limit: number }> {
  const [data, total] = await this.guideRepository.findAndCount({
    where: { userId, deletedAt: IsNull() },
    order: { createdAt: 'DESC' },
    skip: (page - 1) * limit,
    take: limit,
  });
  
  return { data, total, page, limit };
}

// 搜索攻略（标题和内容）
async search(
  keyword: string, 
  page: number = 1, 
  limit: number = 10
): Promise<{ data: Guide[]; total: number }> {
  const query = this.guideRepository
    .createQueryBuilder('guide')
    .where('guide.deletedAt IS NULL')
    .andWhere(
      '(guide.title LIKE :keyword OR guide.content LIKE :keyword)',
      { keyword: `%${keyword}%` }
    )
    .orderBy('guide.createdAt', 'DESC')
    .skip((page - 1) * limit)
    .take(limit);
    
  const [data, total] = await query.getManyAndCount();
  return { data, total };
}

// 增加浏览次数
async incrementViewCount(id: string): Promise<void> {
  await this.guideRepository.increment({ id }, 'viewCount', 1);
}

// 权限验证
private async checkOwnership(guideId: string, userId: string): Promise<void> {
  const guide = await this.guideRepository.findOne({ 
    where: { id: guideId } 
  });
  
  if (!guide) {
    throw new NotFoundException('攻略不存在');
  }
  
  if (guide.userId !== userId) {
    throw new ForbiddenException('无权操作此攻略');
  }
}
```

```python
import logging
from typing import AsyncGenerator, List, Dict, Optional
from openai import AsyncOpenAI, OpenAIError
import asyncio

logger = logging.getLogger(__name__)

class OpenAIService:
    def __init__(self, api_key: str):
        self.client = AsyncOpenAI(api_key=api_key)
        self.max_retries = 3
        self.timeout = 30
        
    async def generate_guide(
        self,
        destination: str,
        days: int,
        budget: float,
        interests: List[str],
        travel_type: str = "休闲游",
        companions: str = "独自",
        **kwargs
    ) -> AsyncGenerator[str, None]:
        """
        生成旅游攻略（流式输出）
        
        Args:
            destination: 目的地
            days: 天数
            budget: 预算
            interests: 兴趣列表
            travel_type: 旅行类型
            companions: 同行人
            
        Yields:
            str: 流式输出的攻略内容
        """
        from app.core.prompts import build_travel_prompt
        
        prompt = build_travel_prompt(
            destination=destination,
            days=days,
            budget=budget,
            interests=interests,
            travel_type=travel_type,
            companions=companions
        )
        
        messages = [{"role": "user", "content": prompt}]
        
        async for chunk in self.stream_chat(messages):
            yield chunk
    
    async def stream_chat(
        self,
        messages: List[Dict[str, str]],
        model: str = "gpt-3.5-turbo",
        temperature: float = 0.7,
    ) -> AsyncGenerator[str, None]:
        """
        流式对话
        
        Args:
            messages: 消息历史（最多10条）
            model: 模型名称
            temperature: 温度参数
            
        Yields:
            str: 流式输出的内容
        """
        # 限制消息数量
        if len(messages) > 10:
            messages = messages[-10:]
            logger.warning(f"消息数量超过10条，已截取最后10条")
        
        retry_count = 0
        while retry_count < self.max_retries:
            try:
                logger.info(f"开始生成，模型: {model}, 消息数: {len(messages)}")
                
                stream = await self.client.chat.completions.create(
                    model=model,
                    messages=messages,
                    temperature=temperature,
                    stream=True,
                    timeout=self.timeout
                )
                
                total_tokens = 0
                async for chunk in stream:
                    if chunk.choices[0].delta.content:
                        content = chunk.choices[0].delta.content
                        total_tokens += len(content)
                        yield content
                
                logger.info(f"生成完成，总字符数: {total_tokens}")
                break
                
            except OpenAIError as e:
                retry_count += 1
                logger.error(f"OpenAI错误 (重试 {retry_count}/{self.max_retries}): {str(e)}")
                
                if retry_count >= self.max_retries:
                    raise Exception(f"AI服务错误: {str(e)}")
                    
                await asyncio.sleep(2 ** retry_count)  # 指数退避
                
            except asyncio.TimeoutError:
                retry_count += 1
                logger.error(f"请求超时 (重试 {retry_count}/{self.max_retries})")
                
                if retry_count >= self.max_retries:
                    raise Exception("AI服务超时，请稍后重试")
                    
                await asyncio.sleep(2 ** retry_count)
                
            except Exception as e:
                logger.error(f"未知错误: {str(e)}")
                raise Exception(f"服务异常: {str(e)}")
```
